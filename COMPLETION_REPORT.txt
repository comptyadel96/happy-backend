Template Godot minimal:

    extends Node
    
    const BACKEND_URL = "http://localhost:3000"
    var auth_token = ""
    
    func _ready():
        # 1. S'inscrire
        register()
        
        # 2. Jouer
        collect_item(1, "chocolate", 0)
        
        # 3. Compléter
        complete_level(1, 250, 180)
    
    func register():
        var result = await POST("/auth/register", {...})
        auth_token = result["token"]
    
    func collect_item(level_id, item_type, item_index):
        await PATCH_AUTH("/game/item-collect", {...})
    
    func complete_level(level_id, score, time):
        var result = await PATCH_AUTH("/game/level-complete", {...})
        print("Score total: ", result["totalScore"])
