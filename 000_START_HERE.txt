📊 APIs DISPONIBLES
─────────────────────────────────────────────────────────────────────────

Authentication:
   POST   /auth/register              ← S'inscrire (adulte ou enfant)
   POST   /auth/login                 ← Se connecter

Game Management:
   GET    /game/profile               ← Récupérer le profil complet
   GET    /game/stats                 ← Obtenir les statistiques
   GET    /game/level/:levelId        ← Infos d'un niveau

Gameplay:
   PATCH  /game/item-collect          ← Collecter chocolat/œuf
   PATCH  /game/level-complete        ← Fin de niveau + score

Synchronisation:
   PATCH  /game/sync                  ← Sync offline→online

Health Check:
   GET    /health/redis               ← Vérifier Redis


🎮 EXEMPLE DE CODE GODOT
─────────────────────────────────────────────────────────────────────────

extends Node

const BACKEND_URL = "http://localhost:3000"
var auth_token = ""

func _ready():
    # 1️⃣ S'inscrire (adulte)
    var reg = await _post("/auth/register", {
        "email": "joueur@example.com",
        "password": "SecurePass123!",
        "fullName": "Jean Dupont",
        "age": 25,
        "phone": "+33612345678",
        "physicalAddress": "123 Rue de Paris"
    })
    auth_token = reg["token"]
    print("✅ Inscrit!")
    
    # 2️⃣ Obtenir les stats
    var stats = await _get("/game/stats")
    print("🏆 Score: ", stats["stats"]["totalScore"])
    print("🍫 Chocolats: ", stats["stats"]["totalChocolates"])
    print("🥚 Œufs: ", stats["stats"]["totalEggs"])
    
    # 3️⃣ Collecter des items
    for i in range(3):
        await _patch("/game/item-collect", {
            "levelId": 1,
            "itemType": "chocolate",
            "itemIndex": i
        })
    
    # 4️⃣ Compléter le niveau
    var result = await _patch("/game/level-complete", {
        "levelId": 1,
        "score": 250,
        "timeSpent": 180
    })
    print("🎉 Score total: ", result["totalScore"])

func _post(ep: String, body: Dictionary):
    var h = HTTPRequest.new()
    add_child(h)
    h.request(BACKEND_URL + ep, [], HTTPClient.METHOD_POST, JSON.stringify(body))
    var r = await h.request_completed
    return JSON.parse_string(r[3].get_string_from_utf8())

func _get(ep: String):
    var h = HTTPRequest.new()
    add_child(h)
    h.request(BACKEND_URL + ep, ["Authorization: Bearer " + auth_token])
    var r = await h.request_completed
    return JSON.parse_string(r[3].get_string_from_utf8())

func _patch(ep: String, body: Dictionary):
    var h = HTTPRequest.new()
    add_child(h)
    var headers = ["Authorization: Bearer " + auth_token, "Content-Type: application/json"]
    h.request(BACKEND_URL + ep, headers, HTTPClient.METHOD_PATCH, JSON.stringify(body))
    var r = await h.request_completed
    return JSON.parse_string(r[3].get_string_from_utf8())
